    <script>
        let turno = 'prima';
        let primaScelte = [];
        let secondaScelte = [];
        let risposteEroi = {
            prima: '',
            seconda: ''
        };

        const bandiere = {
            marocco: 'https://flagpedia.net/data/flags/h80/ma.png',
            spagna: 'https://flagpedia.net/data/flags/h80/es.png',
            francia: 'https://flagpedia.net/data/flags/h80/fr.png',
            gran_bretagna: 'https://flagpedia.net/data/flags/h80/gb.png',
            germania: 'https://flagpedia.net/data/flags/h80/de.png',
            italia: 'https://flagpedia.net/data/flags/h80/it.png',
            austria: 'https://flagpedia.net/data/flags/h80/at.png',
            ottomano: 'https://flagpedia.net/data/flags/h80/tr.png',
            russia: 'https://flagpedia.net/data/flags/h80/ru.png',
            svezia: 'https://flagpedia.net/data/flags/h80/se.png'
        };

        function iniziaGioco() {
            document.getElementById('inizio').style.display = 'none';
            document.getElementById('scelta').style.display = 'block';
            aggiornaTurno();
            aggiungiBandiereAllaLista(); // Aggiungi le bandiere alla lista
        }

        function aggiornaTurno() {
            const turnoElement = document.getElementById('turno');
            turnoElement.textContent = (turno === 'prima') ? "Turno della Prima Squadra" : "Turno della Seconda Squadra";
        }

        function aggiungiBandiereAllaLista() {
            const listaElement = document.getElementById('nazioni-list');
            const nazioni = [
                'marocco', 'spagna', 'francia', 'gran_bretagna', 
                'germania', 'italia', 'austria', 'ottomano', 'russia', 'svezia'
            ];

            nazioni.forEach(nazione => {
                const li = document.createElement('li');
                li.classList.add('nazione-option');
                li.innerHTML = `<img src="${bandiere[nazione]}" alt="Bandiera di ${nazione}" class="bandiera"> ${nazione.charAt(0).toUpperCase() + nazione.slice(1).replace('_', ' ')}`;
                li.onclick = () => selezionaNazione(nazione, li);
                listaElement.appendChild(li);
            });
        }

        function selezionaNazione(nazione, element) {
            if (primaScelte.includes(nazione) || secondaScelte.includes(nazione)) {
                alert("Questa nazione è già stata selezionata!");
                return;
            }

            if (turno === 'prima') {
                primaScelte.push(nazione);
            } else {
                secondaScelte.push(nazione);
            }

            // Disabilita la nazione selezionata
            element.classList.add('disabled');
            element.onclick = null; // Disabilita il click su questa nazione

            turno = (turno === 'prima') ? 'seconda' : 'prima';
            aggiornaTurno();

            if (primaScelte.length === 5 && secondaScelte.length === 5) {
                mostraRiepilogo();
            }
        }

        function mostraRiepilogo() {
            document.getElementById('scelta').style.display = 'none';
            document.getElementById('riepilogo').style.display = 'block';

            const ulPrima = document.getElementById('riepilogo-prima');
            const ulSeconda = document.getElementById('riepilogo-seconda');

            primaScelte.forEach(id => {
                const div = document.createElement('div');
                div.classList.add('riepilogo-item');
                div.innerHTML = `<img src="${bandiere[id]}" class="bandiera" alt="Bandiera di ${id}"> ${id.charAt(0).toUpperCase() + id.slice(1).replace('_', ' ')}`;
                ulPrima.appendChild(div);
            });

            secondaScelte.forEach(id => {
                const div = document.createElement('div');
                div.classList.add('riepilogo-item');
                div.innerHTML = `<img src="${bandiere[id]}" class="bandiera" alt="Bandiera di ${id}"> ${id.charAt(0).toUpperCase() + id.slice(1).replace('_', ' ')}`;
                ulSeconda.appendChild(div);
            });
        }

        function chiediEroi() {
            document.getElementById('riepilogo').style.display = 'none';
            document.getElementById('finestra-eroi').style.display = 'block';
            document.getElementById('risposte').innerHTML = ` 
                <h3>Prima squadra: desiderate utilizzare gli Eroi?</h3>
                <button onclick="scegliEroi('si', 'prima')">Sì</button>
                <button onclick="scegliEroi('no', 'prima')">No</button>
            `;
        }

        function scegliEroi(scelta, squadra) {
            risposteEroi[squadra] = scelta;
            if (risposteEroi.prima && risposteEroi.seconda) {
                mostraRisposteEroi();
            } else {
                // Passiamo alla seconda squadra
                if (squadra === 'prima') {
                    document.getElementById('risposte').innerHTML = ` 
                        <h3>Seconda squadra: desiderate utilizzare gli Eroi?</h3>
                        <button onclick="scegliEroi('si', 'seconda')">Sì</button>
                        <button onclick="scegliEroi('no', 'seconda')">No</button>
                    `;
                }
            }
        }

        function mostraRisposteEroi() {
            document.getElementById('finestra-eroi').style.display = 'none';
            document.getElementById('finestra-fine').style.display = 'block';
        }

        function scaricaPDF() {
            const { jsPDF } = window.jspdf;
            const doc = new jsPDF();

            doc.text('Riepilogo delle scelte', 10, 10);
            doc.text('Prima Squadra:', 10, 20);
            primaScelte.forEach((nazione, index) => {
                doc.text(`${index + 1}. ${nazione.charAt(0).toUpperCase() + nazione.slice(1)} - ${risposteEroi.prima === 'si' ? 'Eroi Attivi' : 'Eroi Non Attivi'}`, 10, 30 + index * 10);
            });

            doc.text('Seconda Squadra:', 10, 30 + primaScelte.length * 10);
            secondaScelte.forEach((nazione, index) => {
                doc.text(`${index + 1}. ${nazione.charAt(0).toUpperCase() + nazione.slice(1)} - ${risposteEroi.seconda === 'si' ? 'Eroi Attivi' : 'Eroi Non Attivi'}`, 10, 40 + (primaScelte.length + index) * 10);
            });

            doc.save('scelte_nazioni_supremacy1914.pdf');
        }
